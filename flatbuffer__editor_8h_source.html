32 /// An on-screen representation of a Flatbuffer, which can be edited by the
33 /// user. Instantiate FlatbufferEditor with a Flatbuffer you'd like it to
34 /// edit. It will create a copy of that Flatbuffer to manipulate, and you can
35 /// get the modified Flatbuffer data back out whenever you want.
36 ///
37 /// When you are ready to get the Flatbuffer data from the editor, you can call
38 /// one of the GetFlatbufferCopy() functions, which will return the data in
39 /// different formats. These functions will copy the data out of our internal
40 /// buffer using reflection (which also has the side effect of normalizing the
41 /// Flatbuffer if we have made changes to it).
42 class FlatbufferEditor {
43  public:
44  /// Construct a FlatbufferEditor for a given schema and table definition,
45  /// and optionally initialize it with flatbuffer data (or nullptr).
46  ///
47  /// When you create a FlatbufferEditor, we will use reflection to copy the
48  /// flatbuffer into our own internal buffer. If you want to change the
49  /// Flatbuffer externally, call SetNewFlatbuffer() and pass in the new
50  /// contents. If you don't have the data you want to edit yet, you can pass in
51  /// nullptr, which means HasFlatbufferData() will be false and can use
52  /// SetFlatbufferData later on. (And if you don't have a
53  /// FlatbufferEditorConfig, just pass in nullptr and we will use default UI
54  /// settings.)
55  FlatbufferEditor(const FlatbufferEditorConfig* config,
56  const reflection::Schema& schema,
57  const reflection::Object& table_def,
58  const void* flatbuffer_data);
59 
60  /// Override the current Flatbuffer data with this new one. Uses reflection to
61  /// copy into our own internal buffers. Will discard whatever is already in
62  /// our copy of the Flatbuffer and in the edit fields.
63  void SetFlatbufferData(const void* flatbuffer_data) {
64  ClearEditFields();
65  ClearFlatbufferModifiedFlag();
66  if (flatbuffer_data != nullptr) {
67  CopyTable(flatbuffer_data, &flatbuffer_);
68  } else {
69  flatbuffer_.clear();
70  }
71  }

204  /// How VisitFlatbuffer* should traverse the table.
205  /// kCheckEdits: Traverse and check if fields have changed, but don't commit
206  /// any changes.
207  /// kDraw*: Draw the Flatbuffer. ReadOnly means use Labels instead of Edit
208  /// fields. Manual means use Edit fields, but require the user to explicitly
209  /// save them back out to the Flatbuffer. Auto means automatically commit the
210  /// values into the Flatbuffer as you edit them.
211  /// kCommitEdits: Traverse, and if fields have changed, commit them to the
212  /// Flatbuffer.
213  enum VisitMode {
214  kCheckEdits, // Only check if any fields have been modified.
215  kDrawEditAuto, // Draw using edit fields that auto-update the FB.
216  kDrawEditManual, // Draw using edit fields that manually update the FB.
217  kDrawReadOnly, // Draw using label fields, not editable
218  kCommitEdits // Write out edits to the Flatbuffer.
219  };

242  /// Visit a single field with the given name and value. The "id" should
243  /// uniquely identify it in the tree of data structures.
244  ///
245  /// If mode is kDrawEdit, the Flatbuffer field will be drawn using a FlatUI
246  /// Edit() control. kDrawReadOnly is similar, but it will use a Label()
247  /// control
248  /// and won't be editable.
249  ///
250  /// If mode is kEdit, it won't actually draw, but it apply any edits made from
251  /// the previously visited fields, hitting all of the previously calculated
252  /// IDs to apply the edits.
253  ///
254  /// Only returns true if mode = kEdit and if we applied any edits.
255  bool VisitField(VisitMode mode, const std::string& name,
256  const std::string& value, const std::string& type,
257  const std::string& comment, const std::string& id);
258 
259  /// Visit a subtable with the given name. The "id" should uniquely identify it
260  /// in the tree of data structures. If mode is kEdit, don't actually draw, but
261  /// do still propagate through the tree of data structures so we can apply
262  /// edits to the previously determined IDs. Returns true if kEdit and if we
263  /// applied any edits.
264  bool VisitSubtable(VisitMode mode, const std::string& field,
265  const std::string& type, const std::string& comment,
