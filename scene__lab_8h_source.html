51  /// Initialize Scene Lab once, when starting your game.
52  ///
53  /// Call this function as soon as you have an entity manager and font
54  /// manager. Consider giving Scene Lab a camera via SetCamera() as well.
55  void Initialize(const SceneLabConfig* config,

59  /// Give Scene Lab a camera that it can use. If you don't call this, it will
60  /// create its own BasicCamera instead.
61  ///
62  /// Scene Lab will take over ownership of `camera`.
63  void SetCamera(std::unique_ptr<corgi::CameraInterface> camera) {
64  camera_ = std::move(camera);
65  }
66 
67  /// While Scene Lab is active, you must call this once a frame, every frame.
68  void AdvanceFrame(corgi::WorldTime delta_time);
69 
70  /// Render Scene Lab and its GUI; only call this when Scene Lab is active.
71  ///
72  /// While Scene Lab is running, you are still responsible for rendering your
73  /// own game world. Call GetCamera() to get the camera you should use for
74  /// rendering.
75  ///
76  /// Warning: if you are actively using FlatUI elsewhere in your code while
77  /// Scene Lab is running, you will need to modify this function to not render
78  /// the GUI here, and call EditorGui::StartRender(), EditorGui::DrawGui(), and
79  /// EditorGui::FinishRender() yourself.
80  void Render(fplbase::Renderer* renderer);
81 
82  /// Activate Scene Lab. Once you call this, you should start calling
83  /// AdvanceFrame and Render each frame, and stop calling
84  /// EntityManager::UpdateComponents() yourself.
85  void Activate();
86 
87  /// Immediately deactivate Scene Lab. The preferred way to exit the editor is
88  /// to use RequestExit, however, as that will give the user a chance to save
89  /// their changes to the world.
90  void Deactivate();
91 
92  /// When you activate the editor, you can pass in your game's camera so the
93  /// user can seamlessly be positioned at the same place they were during the
94  /// game. Note that if you do not yet have a Scene Lab camera set when you
95  /// call this method, a BasicCamera will be created for you.
96  ///
97  /// You can also set the initial camera by calling GetCamera()->set_position()
98  /// and/or GetCamera()->set_facing().
99  void SetInitialCamera(const corgi::CameraInterface& initial_camera);
100 
101  /// Get the Scene Lab camera, so you can render the scene properly
102  /// or change its position. If you do not have a camera, a
103  /// BasicCamera will be created for you.
104  corgi::CameraInterface* GetCamera();
105 
106  /// Highlight the specified entity, so that you can change its properties.
107  void SelectEntity(const corgi::EntityRef& entity_ref);
108 
109  /// Save the current positions and properties of all entities.
110  ///
111  /// If `to_disk` is true, save to .bin and .json files and update the entity
112  /// factory's file cache. Otherwise, just update the file cache but don't
113  /// physically save the files to disk.
114  ///
115  /// If you are saving to disk, entities will be saved to the files they were
116  /// initially loaded from.
117  void SaveScene(bool to_disk);
118 
119  /// Save the current positions and properties to disk.
120  ///
121  /// See SaveScene(bool to_disk) for more details.
122  void SaveScene() { SaveScene(true); }
123 
124  /// Save all the entities that were from a specific file to that file on disk.
125  ///
126  /// Called by SaveScene() when saving to disk, but you could always call
127  /// this directly.
128  void SaveEntitiesInFile(const std::string& filename);
129 
130  /// Request that Scene Lab exit.
131  ///
132  /// If you haven't saved your changes, it will prompt you to do so, keep them
133  /// in memory, or abandon them. Once Scene Lab decides it's okay to exit,
134  /// IsReadyToExit() will return true.
135  ///
136  /// After you've exited, you can always get back into Scene Lab by calling
137  /// Activate() again.
138  void RequestExit();
139 
140  /// Abort a previously-requested exit, which hides the confirmation dialog.
141  void AbortExit();
142 
143  /// Returns true if we are ready to exit Scene Lab (everything is saved or
144  /// discarded, etc), or false if not. Once it returns true, you can safely
145  /// deactivate the editor.
146  bool IsReadyToExit();
147 
148  /// Add a component to the list of components Scene Lab updates each frame.
149  ///
150  /// While Scene Lab is activated, you should no longer be calling
151  /// EntityManager::UpdateComponents(); you should let Scene Lab update only
152  /// the components it cares about. If you have any components you are sure you
153  /// also want updated while editing the scene, add them to the list by calling
154  /// this function.
155  void AddComponentToUpdate(corgi::ComponentId component_id) {
156  components_to_update_.push_back(component_id);
157  }
158 
159  /// Externally mark that some entities have been modified.
160  ///
161  /// Generally used by Scene Lab's GUI, but if you change an entity's
162  /// properties some other way, call this function to ensure the user will be
163  /// prompted to save on exiting the editor.
164  void set_entities_modified(bool b) { entities_modified_ = b; }
165 
166  /// Have entities been modified? If so, prompt the user to save before exit.
167  bool entities_modified() const { return entities_modified_; }
