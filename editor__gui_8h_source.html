35 /// EditorGui is a FlatUI-based GUI overlay for Scene Lab to the edit
36 /// window and on-screen menu buttons. Scene Lab can work without the
37 /// GUI but will have very reduced functionality.
38 class EditorGui {
39  public:
40  /// Initialize the user interface. Ensure that you pass the Flatbuffers schema
41  /// you are using (in binary schema format) so FlatbufferEditor knows how to
42  /// present the data. EditorGui will load its own font (specified in the
43  /// config) into an existing FontManager.
44  EditorGui(const SceneLabConfig* config, SceneLab* scene_lab,
45  corgi::EntityManager* entity_manager,
46  flatui::FontManager* font_manager, const std::string* schema_data);
47 
48  /// Turn on the GUI. This should be called when Scene Lab is activated.
49  void Activate();
50 
51  /// Turn off the GUI. This should be called when Scene Lab is deactivated.
52  void Deactivate() {}
53 
54  /// Render the GUI, then update based on button presses. You must either call
55  /// this, or call StartRender(), then DrawGui() in a FlatUI gui::Run context,
56  /// then FinishRenderRender(). (only DrawGui() should be inside a FlatUI
57  /// gui::Run context.)
58  void Render();
59 
60  /// Prepare to draw the GUI. You don't need to call this if you are using
61  /// Render(). See DrawGui() for more information.
62 
63  void StartRender();
64  /// Either call Render() above, or call the following sequence if you are
65  /// using FlatUI elsewhere:
66  /// * StartRender();
67  /// * gui::Run(() { ... other Gui... DrawGui(...) };
68  /// * FinishRender();
69  void DrawGui(const mathfu::vec2& virtual_resolution);
70 
71  /// Finish drawing the GUI, and update input. You don't need to call this if
72  /// you are using Render(). See DrawGui() for more information.
73  void FinishRender();
74 
75  /// Is the GUI capturing mouse and/or keyboard input? Returns true if you
76  /// should ignore mouse and keyboard input. Or use mouse_in_window() and
77  /// keyboard_in_use() for more fine-grained control.
78  bool InputCaptured() const { return mouse_in_window() || keyboard_in_use(); }
79 
80  /// Is the mouse pointer over a GUI element? If true, you should ignore mouse
81  /// clicks as the user is clicking on GUI elements instead.
82  bool mouse_in_window() const { return mouse_in_window_; }
83 
84  /// If we've taken ownership of the keyboard (generally when a editable field
85  /// is being edited), this will return true, telling you to ignore keyboard
86  /// input.
87  bool keyboard_in_use() const { return keyboard_in_use_; }
88 
89  /// Should camera movement in Scene Lab be parallel to the ground, so your
90  /// camera's height doesn't change unless you want it to?
91  bool lock_camera_height() const { return lock_camera_height_; }
92 
93  /// Choose which entity we are currently editing.
94  void SetEditEntity(corgi::EntityRef& entity);
95  /// Get the entity we are currently editing.
96  corgi::EntityRef edit_entity() const { return edit_entity_; }
97 
98  /// Clear all cached or modified data that we have for the edit entity. Call
99  /// this if you change any entity data externally, so we can reload data
100  /// directly from the entity.
101  void ClearEntityData() { component_guis_.clear(); }
102  /// Write the entity fields that were changed to the entity_data_ flatbuffer,
103  /// then import the changed flatbuffers back to the edit_entity_.
104  void CommitEntityData();
105 
106  /// Should we let the editor exit? Not if there are pending changes.
107  bool CanExit();
108 
109  /// Should we let the editor deselect this as the current entity?
110  /// Probably not if we've made any changes.
111  bool CanDeselectEntity() const { return !EntityModified(); }
112 
113  /// Has the entity's flatbuffer been modified?
114  bool EntityModified() const {
115  for (auto iter = component_guis_.begin(); iter != component_guis_.end();
116  ++iter) {
117  FlatbufferEditor* editor = iter->second.get();
118  if (editor->flatbuffer_modified()) return true;
119  }
120  return false;
121  }
122 
123  /// "Entity Updated" callback for Scene Lab; if the entity is updated
124  /// externally, we reload its data by calling ClearEntityData().
125  void EntityUpdated(corgi::EntityRef entity);
126 
127  /// Does the user want you to show the current entity's physics?
128  bool show_physics() const { return show_physics_; }
129 
130  /// Returns which mouse mode index we have selected.
131  unsigned int mouse_mode_index() const { return mouse_mode_index_; }
132  /// Set which mouse mode index the user wants to use.
133  void set_mouse_mode_index(int m) { mouse_mode_index_ = m; }
134 
135  const std::string& menu_title_string() { return menu_title_string_; }
136  void set_menu_title_string(const std::string& s) { menu_title_string_ = s; }
